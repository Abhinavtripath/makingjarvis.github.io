print(" ")

def takecommand():
r = sr.Recognizer()
with sr.Microphone() as source:
print("Listening....")
r.pause_threshold = 1

try:
print("Recognizing....")
query = r.recognize_google(audio, language='en-in')
print(f"you said: {query}\n")
except Exception as e:
print(e)
return "None"
return query
